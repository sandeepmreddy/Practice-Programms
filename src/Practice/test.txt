CREATE PROCEDURE DeleteRows
    @TableName NVARCHAR(50),
    @ColumnName NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @sql NVARCHAR(MAX);
    DECLARE @id INT;

    DECLARE cur CURSOR FOR
    SELECT Id FROM @TableName WHERE @ColumnName IS NOT NULL;
    OPEN cur;

    FETCH NEXT FROM cur INTO @id;

    WHILE @@FETCH_STATUS = 0
    BEGIN
        SET @sql = 'DELETE FROM ' + @TableName + ' WHERE Id = ' + CAST(@id AS NVARCHAR(10)) + ';';
        EXEC sp_executesql @sql;
        FETCH NEXT FROM cur INTO @id;
    END

    CLOSE cur;
    DEALLOCATE cur;
END
